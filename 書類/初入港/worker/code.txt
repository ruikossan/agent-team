function onOpen() {
  SpreadsheetApp.getUi().createMenu(APP_CONFIG.menuName)
    .addItem('入力フォーム', 'openInputDialog')
    .addItem('下書きPDF作成', 'exportDraftPdfFromSheet')
    .addSeparator()
    .addItem('シートをクリア', 'clearSheet')
    .addToUi();
}

function openInputDialog() {
  const template = HtmlService.createTemplateFromFile('index');
  template.initialState = JSON.stringify(getFormState());
  SpreadsheetApp.getUi().showModalDialog(template.evaluate().setWidth(1160).setHeight(780), '初入港チェック表入力');
}

function getFormState() {
  const sheet = getMainSheet_();
  return { fields: readFields_(sheet), checklist: readChecklist_(sheet), queue: getQueueSummaries_() };
}

function saveForm(payload) {
  writePayloadToSheet_(getMainSheet_(), normalizePayload_(payload));
  SpreadsheetApp.flush();
  return getFormState();
}

function exportDraftPdf(payload) {
  const data = normalizePayload_(payload);
  const sheet = getMainSheet_();
  writePayloadToSheet_(sheet, data);
  validateRequiredFields_(data.fields);
  placeTextStamp_(sheet.getRange(APP_CONFIG.fields.workerStampName[0]), data.fields.workerStampName);
  const caseId = generateCaseId_();
  assertCaseIdDoesNotExistAnywhere_(caseId);
  const file = exportSheetPdfToFolder_(getFolderByIdOrThrow_(APP_CONFIG.draftPdfFolderId), buildPdfName_(data.fields, caseId));
  const record = { caseId, status: APP_CONFIG.initialOverallStatus, payload: data, draftPdfUrl: file.getUrl(), draftPdfName: file.getName(), finalPdfUrl: '', finalPdfName: '' };
  upsertQueueRecord_(record);
  syncMenuDocumentsRecord_(record);
  syncMenuStaffRecord_(record);
  syncManagerMirrorRecord_(record);
  return { caseId: record.caseId, pdfName: record.draftPdfName, queue: getQueueSummaries_() };
}

function exportDraftPdfFromSheet() {
  return exportDraftPdf({ fields: readFields_(getMainSheet_()), checklist: readChecklist_(getMainSheet_()) });
}

function clearSheet() {
  const ui = SpreadsheetApp.getUi();
  if (ui.alert('確認', '入力内容をクリアしますか。', ui.ButtonSet.YES_NO) !== ui.Button.YES) return;
  const sheet = getMainSheet_();
  Object.keys(APP_CONFIG.fields).forEach((key) => sheet.getRange(APP_CONFIG.fields[key][0]).clearContent());
  APP_CONFIG.checklist.forEach((item) => ['checkCell', 'remarkCell', 'actionCell'].forEach((k) => sheet.getRange(item[k]).clearContent()));
}

function normalizePayload_(payload) {
  const fields = {};
  Object.keys(APP_CONFIG.fields).forEach((key) => fields[key] = String(payload && payload.fields && payload.fields[key] || '').trim());
  const checklist = APP_CONFIG.checklist.map((item) => {
    const source = Array.isArray(payload && payload.checklist) ? payload.checklist.find((entry) => entry && entry.key === item.key) : null;
    return { key: item.key, check: source ? String(source.check || '').trim() : '', remark: source ? String(source.remark || '').trim() : '', action: source ? String(source.action || '').trim() : '' };
  });
  return { fields, checklist };
}

function validateRequiredFields_(fields) {
  const missing = APP_CONFIG.requiredFields.filter((key) => !fields[key]).map((key) => APP_CONFIG.fields[key][2]);
  if (missing.length) throw new Error('必須項目が未入力です: ' + missing.join('、'));
}

function writePayloadToSheet_(sheet, payload) {
  Object.keys(APP_CONFIG.fields).forEach((key) => writeValue_(sheet.getRange(APP_CONFIG.fields[key][0]), payload.fields[key], APP_CONFIG.fields[key][1]));
  APP_CONFIG.checklist.forEach((item) => {
    const row = payload.checklist.find((entry) => entry.key === item.key) || { check: '', remark: '', action: '' };
    sheet.getRange(item.checkCell).setValue(row.check);
    sheet.getRange(item.remarkCell).setValue(row.remark);
    sheet.getRange(item.actionCell).setValue(row.action);
  });
}

function readFields_(sheet) {
  const fields = {};
  Object.keys(APP_CONFIG.fields).forEach((key) => fields[key] = readValue_(sheet.getRange(APP_CONFIG.fields[key][0]), APP_CONFIG.fields[key][1]));
  return fields;
}

function readChecklist_(sheet) {
  return APP_CONFIG.checklist.map((item) => ({ key: item.key, label: item.label, check: String(sheet.getRange(item.checkCell).getDisplayValue() || '').trim(), remark: String(sheet.getRange(item.remarkCell).getDisplayValue() || '').trim(), action: String(sheet.getRange(item.actionCell).getDisplayValue() || '').trim() }));
}

function writeValue_(range, value, type) {
  if (type === 'date') {
    if (!value) return range.clearContent();
    const date = new Date(value);
    if (Number.isNaN(date.getTime())) throw new Error('日付の形式が不正です: ' + value);
    range.setValue(date).setNumberFormat('yyyy"年"m"月"d"日"');
    return;
  }
  range.setValue(value || '');
}

function readValue_(range, type) {
  if (type === 'date') {
    const value = range.getValue();
    return value instanceof Date && !Number.isNaN(value.getTime()) ? Utilities.formatDate(value, Session.getScriptTimeZone(), 'yyyy-MM-dd') : '';
  }
  return String(range.getDisplayValue() || '').trim();
}

function placeTextStamp_(range, value) {
  range.setValue(value || '').setHorizontalAlignment('center').setVerticalAlignment('middle').setFontWeight('bold').setFontSize(14);
}

function generateCaseId_() {
  const lock = LockService.getScriptLock();
  lock.waitLock(10000);
  try {
    const ss = getCaseIdSpreadsheet_();
    const sheet = getCaseIdSheet_(ss);
    const dataStartRow = getCaseIdDataStartRow_(sheet);
    const caseId = findNextCaseIdRecordId_(sheet, dataStartRow);
    reserveCaseIdRecord_(sheet, dataStartRow, caseId);
    return caseId;
  } finally {
    lock.releaseLock();
  }
}

function buildPdfName_(fields, caseId) {
  return [caseId, String(fields.shipName || '').trim(), APP_CONFIG.documentType, '下書き'].filter(Boolean).join(' ');
}

function exportSheetPdfToFolder_(folder, baseName) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const url = 'https://docs.google.com/spreadsheets/d/' + ss.getId() + '/export?format=pdf&gid=' + getMainSheet_().getSheetId() + '&range=' + encodeURIComponent(APP_CONFIG.pdfRangeA1) + '&size=A4&portrait=true&fitw=true&sheetnames=false&printtitle=false&pagenumbers=false&gridlines=false&fzr=false';
  const blob = UrlFetchApp.fetch(url, { headers: { Authorization: 'Bearer ' + ScriptApp.getOAuthToken() } }).getBlob().setName(baseName + '.pdf');
  return folder.createFile(blob);
}

function upsertQueueRecord_(record) {
  const sheet = ensureQueueSheet_();
  const rowIndex = findQueueRowByCaseId_(sheet, record.caseId);
  const duplicateCount = countCaseIdInSheetColumn_(sheet, record.caseId, 1, 2);
  if ((!rowIndex && duplicateCount > 0) || (rowIndex && duplicateCount > 1)) throw new Error('案件IDが重複しています: ' + record.caseId);
  const row = [record.caseId, record.status, JSON.stringify(record.payload), record.draftPdfUrl || '', record.draftPdfName || '', record.finalPdfUrl || '', record.finalPdfName || ''];
  if (rowIndex) sheet.getRange(rowIndex, 1, 1, row.length).setValues([row]); else sheet.appendRow(row);
}

function ensureQueueSheet_() {
  let sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(APP_CONFIG.queueSheetName);
  if (!sheet) {
    sheet = SpreadsheetApp.getActiveSpreadsheet().insertSheet(APP_CONFIG.queueSheetName);
    sheet.getRange(1, 1, 1, 7).setValues([['caseId','status','payloadJson','draftPdfUrl','draftPdfName','finalPdfUrl','finalPdfName']]);
    sheet.hideSheet();
  }
  return sheet;
}

function findQueueRowByCaseId_(sheet, caseId) {
  const lastRow = sheet.getLastRow();
  if (lastRow < 2) return 0;
  const values = sheet.getRange(2, 1, lastRow - 1, 1).getDisplayValues();
  for (let i = 0; i < values.length; i += 1) if (String(values[i][0] || '').trim() === String(caseId || '').trim()) return i + 2;
  return 0;
}

function getQueueSummaries_() {
  const sheet = ensureQueueSheet_();
  const lastRow = sheet.getLastRow();
  if (lastRow < 2) return [];
  return sheet.getRange(2, 1, lastRow - 1, 3).getValues().map((row) => {
    const payload = row[2] ? JSON.parse(row[2]) : { fields: {} };
    return { caseId: row[0], status: row[1], shipName: payload.fields.shipName || '', caseContent: buildCaseContent_(payload.fields || {}) };
  }).filter((item) => item.caseId);
}

function syncMenuDocumentsRecord_(record) {
  if (!isConfiguredId_(APP_CONFIG.menuSpreadsheetId)) return;
  const sheet = SpreadsheetApp.openById(APP_CONFIG.menuSpreadsheetId).getSheetByName(APP_CONFIG.menuDocumentsSheetName);
  assertCaseIdUniqueInSheet_(sheet, record.caseId, '各種書類シート');
  const rowIndex = findOrCreateRecordRow_(sheet, record.caseId);
  sheet.getRange(rowIndex, 1).setValue(record.caseId);
  sheet.getRange(rowIndex, APP_CONFIG.transferColumn).setValue(buildCaseContent_(record.payload.fields));
  sheet.getRange(rowIndex, APP_CONFIG.transferColumn + APP_CONFIG.statusOffset).setValue(APP_CONFIG.pendingDocumentStatus);
}

function syncMenuStaffRecord_(record) {
  if (!isConfiguredId_(APP_CONFIG.menuSpreadsheetId)) return;
  const sheet = SpreadsheetApp.openById(APP_CONFIG.menuSpreadsheetId).getSheetByName(APP_CONFIG.menuStaffSheetName);
  assertCaseIdUniqueInSheet_(sheet, record.caseId, '主務職シート');
  const rowIndex = findOrCreateRecordRow_(sheet, record.caseId);
  sheet.getRange(rowIndex, 1).setValue(record.caseId);
  sheet.getRange(rowIndex, 2).setValue(buildCaseContent_(record.payload.fields));
  sheet.getRange(rowIndex, 3).setFormula(`=HYPERLINK("${escapeFormulaText_(buildManagerSpreadsheetUrl_())}","${escapeFormulaText_(record.caseId)}")`);
  sheet.getRange(rowIndex, 4).setValue(APP_CONFIG.initialRoleStatus);
  sheet.getRange(rowIndex, 5).setValue(APP_CONFIG.initialRoleStatus);
  sheet.getRange(rowIndex, 6).setValue(APP_CONFIG.initialOverallStatus);
}

function syncManagerMirrorRecord_(record) {
  if (!isConfiguredId_(APP_CONFIG.managerSpreadsheetId)) return;
  const ss = SpreadsheetApp.openById(APP_CONFIG.managerSpreadsheetId);
  let sheet = ss.getSheetByName(APP_CONFIG.queueSheetName);
  if (!sheet) {
    sheet = ss.insertSheet(APP_CONFIG.queueSheetName);
    sheet.getRange(1, 1, 1, 7).setValues([['caseId','status','payloadJson','draftPdfUrl','draftPdfName','finalPdfUrl','finalPdfName']]);
    sheet.hideSheet();
  }
  const rowIndex = findQueueRowByCaseId_(sheet, record.caseId);
  const duplicateCount = countCaseIdInSheetColumn_(sheet, record.caseId, 1, 2);
  if ((!rowIndex && duplicateCount > 0) || (rowIndex && duplicateCount > 1)) throw new Error('manager 側キューで案件IDが重複しています: ' + record.caseId);
  const row = [record.caseId, record.status, JSON.stringify(record.payload), record.draftPdfUrl || '', record.draftPdfName || '', '', ''];
  if (rowIndex) sheet.getRange(rowIndex, 1, 1, row.length).setValues([row]); else sheet.appendRow(row);
}

function buildCaseContent_(fields) { return [String(fields.shipName || '').trim(), APP_CONFIG.documentType].filter(Boolean).join(' '); }
function buildManagerSpreadsheetUrl_() { return isConfiguredId_(APP_CONFIG.managerSpreadsheetId) ? 'https://docs.google.com/spreadsheets/d/' + APP_CONFIG.managerSpreadsheetId + '/edit' : ''; }
function findOrCreateRecordRow_(sheet, caseId) { const lastRow = Math.max(sheet.getLastRow(), 2); const values = sheet.getRange(2,1,lastRow-1,1).getDisplayValues(); for (let i=0;i<values.length;i+=1){ if(String(values[i][0]||'').trim()===String(caseId||'').trim() || !String(values[i][0]||'').trim()) return i+2; } return lastRow+1; }
function assertCaseIdDoesNotExistAnywhere_(caseId) {
  const queueCount = countCaseIdInSheetColumn_(ensureQueueSheet_(), caseId, 1, 2);
  if (queueCount > 0) throw new Error('案件IDが既に存在します: ' + caseId);
  if (isConfiguredId_(APP_CONFIG.menuSpreadsheetId)) {
    const ss = SpreadsheetApp.openById(APP_CONFIG.menuSpreadsheetId);
    assertCaseIdUniqueInSheet_(ss.getSheetByName(APP_CONFIG.menuDocumentsSheetName), caseId, '各種書類シート', true);
    assertCaseIdUniqueInSheet_(ss.getSheetByName(APP_CONFIG.menuStaffSheetName), caseId, '主務職シート', true);
  }
  if (isConfiguredId_(APP_CONFIG.caseIdSpreadsheetId)) {
    const sheet = getCaseIdSheet_(getCaseIdSpreadsheet_());
    const count = countCaseIdInSheetColumn_(sheet, caseId, 1, getCaseIdDataStartRow_(sheet));
    if (count > 1) throw new Error('案件IDリストで案件IDが重複しています: ' + caseId);
  }
}
function assertCaseIdUniqueInSheet_(sheet, caseId, label, disallowAny) {
  if (!sheet) return;
  const count = countCaseIdInSheetColumn_(sheet, caseId, 1, 2);
  if ((disallowAny && count > 0) || (!disallowAny && count > 1)) throw new Error(label + ' で案件IDが重複しています: ' + caseId);
}
function countCaseIdInSheetColumn_(sheet, caseId, column, startRow) {
  if (!sheet) return 0;
  const id = String(caseId || '').trim();
  if (!id) return 0;
  const begin = Math.max(startRow || 2, 1);
  const lastRow = sheet.getLastRow();
  if (lastRow < begin) return 0;
  const values = sheet.getRange(begin, column || 1, lastRow - begin + 1, 1).getDisplayValues();
  let count = 0;
  for (let i = 0; i < values.length; i += 1) if (String(values[i][0] || '').trim() === id) count += 1;
  return count;
}
function getCaseIdSpreadsheet_() {
  const spreadsheetId = String(APP_CONFIG.caseIdSpreadsheetId || '').trim();
  if (!isConfiguredId_(spreadsheetId)) throw new Error('案件ID台帳のスプレッドシートIDを設定してください。');
  return SpreadsheetApp.openById(spreadsheetId);
}
function getCaseIdSheet_(ss) {
  const sheet = ss.getSheetByName(APP_CONFIG.caseIdSheetName);
  if (!sheet) throw new Error('案件ID台帳シートが見つかりません。');
  return sheet;
}
function getCaseIdDataStartRow_(sheet) {
  const maxScan = Math.min(sheet.getLastRow(), 200);
  if (maxScan < 1) return 2;
  const values = sheet.getRange(1, 1, maxScan, 2).getDisplayValues();
  for (let i = 0; i < values.length; i += 1) {
    if (String(values[i][0] || '').trim() === '案件ID') return i + 2;
  }
  return 2;
}
function findNextCaseIdRecordId_(sheet, startRow) {
  const datePart = Utilities.formatDate(new Date(), Session.getScriptTimeZone(), 'yyyyMMdd');
  const prefix = 'SHN-' + datePart + '-';
  const escapedPrefix = prefix.replace(/[-/\\^$*+?.()|[\]{}]/g, '\\$&');
  const pattern = new RegExp('^' + escapedPrefix + '(\\d{4})$');
  const lastRow = sheet.getLastRow();
  let nextNumber = 1;
  if (lastRow >= startRow) {
    const values = sheet.getRange(startRow, 1, lastRow - startRow + 1, 1).getDisplayValues();
    for (let i = 0; i < values.length; i += 1) {
      const match = String(values[i][0] || '').trim().match(pattern);
      if (!match) continue;
      nextNumber = Math.max(nextNumber, Number(match[1]) + 1);
    }
  }
  return prefix + String(nextNumber).padStart(4, '0');
}
function reserveCaseIdRecord_(sheet, startRow, caseId) {
  const existingRow = findCaseIdRowByRecordId_(sheet, caseId, startRow);
  if (existingRow > 0) return existingRow;
  const writeRow = findNextBlankCaseIdRow_(sheet, startRow);
  sheet.getRange(writeRow, 1).setValue(caseId);
  return writeRow;
}
function findCaseIdRowByRecordId_(sheet, caseId, startRow) {
  const begin = Math.max(startRow || 2, 2);
  const lastRow = sheet.getLastRow();
  if (lastRow < begin) return 0;
  const values = sheet.getRange(begin, 1, lastRow - begin + 1, 1).getDisplayValues();
  for (let i = 0; i < values.length; i += 1) {
    if (String(values[i][0] || '').trim() === String(caseId || '').trim()) return begin + i;
  }
  return 0;
}
function findNextBlankCaseIdRow_(sheet, startRow) {
  const begin = Math.max(startRow || 2, 2);
  const lastRow = Math.max(sheet.getLastRow(), begin);
  const values = sheet.getRange(begin, 1, lastRow - begin + 1, 1).getDisplayValues();
  for (let i = 0; i < values.length; i += 1) {
    if (!String(values[i][0] || '').trim()) return begin + i;
  }
  return lastRow + 1;
}
