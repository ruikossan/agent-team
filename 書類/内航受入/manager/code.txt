function saveOperationRecord(payload) {
  const sheet = getMainSheet_();
  const data = normalizePayload_(payload);

  if (data.department) {
    applyDepartmentSelection_(sheet, data.department);
    updatePersonValidations_(sheet, data.department);
  }
  writeMappedFields_(sheet, data.fields);
  writeLogEntries_(sheet, data.logs);
  restoreCheckboxStateMap_(sheet, data.checks || {});

  if (data.fields.staffStampName) {
    sheet.getRange(APP_CONFIG.stampSlots.staff.selectorCell).setValue(data.fields.staffStampName);
  }

  SpreadsheetApp.flush();
  return getAppState();
}

function patchOperationRecord(payload) {
  const sheet = getMainSheet_();
  const data = payload || {};

  if (Object.prototype.hasOwnProperty.call(data, 'department')) {
    const department = String(data.department || '').trim();
    if (department) {
      applyDepartmentSelection_(sheet, department);
      updatePersonValidations_(sheet, department);
    } else {
      clearDepartmentSelection_(sheet);
      updatePersonValidations_(sheet, '');
    }
  }

  if (data.fields && typeof data.fields === 'object') {
    Object.keys(data.fields).forEach((key) => {
      if (!APP_CONFIG.fieldMap[key]) return;
      const field = APP_CONFIG.fieldMap[key];
      writeCellByType_(sheet.getRange(field.cell), data.fields[key], field.type);
    });
  }

  if (Array.isArray(data.logs)) {
    const logMap = {};
    data.logs.forEach((entry) => {
      const key = String(entry && entry.key || '').trim();
      if (key) logMap[key] = entry;
    });

    APP_CONFIG.logEntries.forEach((target) => {
      const source = logMap[target.key];
      if (!source) return;
      if (Object.prototype.hasOwnProperty.call(source, 'time')) writeCellByType_(sheet.getRange(target.time), source.time, 'number');
      if (Object.prototype.hasOwnProperty.call(source, 'pressure')) writeCellByType_(sheet.getRange(target.pressure), source.pressure, 'number');
      if (Object.prototype.hasOwnProperty.call(source, 'flow')) writeCellByType_(sheet.getRange(target.flow), source.flow, 'number');
      if (Object.prototype.hasOwnProperty.call(source, 'amount')) writeCellByType_(sheet.getRange(target.amount), source.amount, 'number');
      if (Object.prototype.hasOwnProperty.call(source, 'person')) writeCellByType_(sheet.getRange(target.person), source.person, 'text');
    });
  }

  SpreadsheetApp.flush();
  return getMainSidebarState();
}

function pressStamp(slotKey, stampName) {
  const sheet = getMainSheet_();
  const slot = APP_CONFIG.stampSlots[slotKey];
  assertStampOrderForCurrentRecord_(slotKey);
  if (!slot) throw new Error('印鑑設定が見つかりません。');

  if (slot.mode === 'assigned') {
    placeAssignedStamp_(sheet, slotKey, slot);
  } else {
    const name = String(stampName || '').trim();
    if (!name) throw new Error('主務職印の名前を選択してください。');
    sheet.getRange(slot.selectorCell).setValue(name);
    placeNamedStamp_(sheet, slot, name);
  }

  SpreadsheetApp.flush();
  const nextStatus = syncQueueProgressForCurrentRecord_(slotKey, sheet);
  if (nextStatus === 'ready_final') {
    finalizeCurrentRecordSilently_(getCurrentQueueRecordId_(), sheet);
  }
  hideStampCacheArea_(sheet);
  return getStampSidebarState();
}

function getOilTypeOptions_() {
  const sheet = getSheetByNameOrThrow_(APP_CONFIG.dropdownSettingsSheetName);
  const lastRow = sheet.getLastRow();
  if (lastRow < 1) return [];
  const values = sheet.getRange(1, 1, lastRow, 1).getDisplayValues();
  const seen = {};
  return values
    .map((row) => String(row[0] || '').trim())
    .filter((value) => {
      if (!value || seen[value]) return false;
      seen[value] = true;
      return true;
    });
}

function getShipNameSuggestionState_(sheet) {
  const seen = {};
  const ordered = [];
  const addName = (value) => {
    const name = String(value || '').trim();
    if (!name || seen[name]) return;
    seen[name] = true;
    ordered.push(name);
  };

  addName(readCellForUi_(sheet.getRange(APP_CONFIG.fieldMap.shipName.cell), 'text'));

  const queueSheet = SpreadsheetApp.getActive().getSheetByName(APP_CONFIG.queueSheetName);
  if (queueSheet) {
    const lastRow = queueSheet.getLastRow();
    if (lastRow >= 2) {
      const values = queueSheet.getRange(2, 5, lastRow - 1, 1).getDisplayValues();
      for (let i = values.length - 1; i >= 0; i -= 1) {
        addName(values[i][0]);
      }
    }
  }

  return {
    options: ordered.slice(0, 200),
    quickPicks: ordered.slice(0, 12),
  };
}

function ensureOilTypeOption_(value) {
  const oilType = String(value || '').trim();
  if (!oilType) return false;

  const sheet = getSheetByNameOrThrow_(APP_CONFIG.dropdownSettingsSheetName);
  const lastRow = sheet.getLastRow();
  const values = lastRow > 0 ? sheet.getRange(1, 1, lastRow, 1).getDisplayValues() : [];
  for (let i = 0; i < values.length; i += 1) {
    if (String(values[i][0] || '').trim() === oilType) return false;
  }
  sheet.getRange(lastRow + 1, 1).setValue(oilType);
  return true;
}

function clearOperationSheet() {
  const ui = SpreadsheetApp.getUi();
  const response = ui.alert('確認', '入力内容と表示中の印鑑をクリアしますか？', ui.ButtonSet.YES_NO);
  if (response !== ui.Button.YES) return;

  const sheet = getMainSheet_();
  APP_CONFIG.clearRanges.forEach((rangeA1) => sheet.getRange(rangeA1).clearContent());
  sheet.getRange(1, 3, sheet.getMaxRows(), 1).uncheck();
  resetVisibleStamps_(sheet);
  hideStampCacheArea_(sheet);
  clearDepartmentSelection_(sheet);
  updatePersonValidations_(sheet, '');
  setCurrentQueueRecordId_('');

  ui.alert('完了', '入力内容と印鑑表示を初期状態に戻しました。', ui.ButtonSet.OK);
}

function refreshStampCache() {
  const sheet = getMainSheet_();
  const mapSheet = getSheetByNameOrThrow_(APP_CONFIG.stampMapSheetName);
  const values = mapSheet.getDataRange().getDisplayValues();
  if (values.length < 2) throw new Error('IMG_MAP に印鑑マスタがありません。');

  resetVisibleStamps_(sheet);
  removeImagesOnCacheRow_(sheet);

  const rows = values.slice(1).filter((row) => row[0] && row[1]);
  rows.forEach((row, index) => {
    const name = String(row[0]).trim();
    const fileId = String(row[1]).trim();
    const file = DriveApp.getFileById(fileId);
    const column = APP_CONFIG.stampCacheStartColumn + index;
    const image = sheet.insertImage(file.getBlob(), column, APP_CONFIG.stampCacheRow);
    image.setWidth(APP_CONFIG.stampImageSize).setHeight(APP_CONFIG.stampImageSize);
    image.setAltTextTitle(APP_CONFIG.stampCacheTitle);
    image.setAltTextDescription(name);
  });

  SpreadsheetApp.getActive().toast(`印鑑キャッシュを更新しました: ${rows.length}件`);
  return getStampSidebarState();
}

function getStampMapNames_() {

    .filter((name) => {
      if (!name || seen[name]) return false;
      seen[name] = true;
      return true;
    });
}

function savePdfAndLogLink() {
  return saveDraftPdfAndQueue();
  const ui = SpreadsheetApp.getUi();
  const sheet = getMainSheet_();
  const fields = getCurrentFieldValues_(sheet);
  const missing = APP_CONFIG.requiredForPdf
    .filter((item) => item.key !== 'staffStampName')
    .filter((item) => !String(fields[item.key] || '').trim())
    .map((item) => item.label);

  if (missing.length) {
    ui.alert('入力不足のため中止', '次の項目が未入力です。\n\n・' + missing.join('\n・'), ui.ButtonSet.OK);
    return;
  }

  const confirm = ui.alert('確認', 'PDFを保存し、ログシートへリンクを追加します。よろしいですか？', ui.ButtonSet.OK_CANCEL);
  if (confirm !== ui.Button.OK) return;

  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const folder = getPdfFolder_();
  const baseName = buildPdfBaseName_(fields);
  const fileBase = getUniqueFileName_(folder, baseName);
  const range = sheet.getRange(APP_CONFIG.pdfRangeA1);
  const pdfUrl =
    'https://docs.google.com/spreadsheets/d/' + ss.getId() +
    '/export?format=pdf' +
    '&gid=' + sheet.getSheetId() +
    '&range=' + encodeURIComponent(range.getA1Notation()) +
    '&size=A4' +
    '&portrait=true' +
    '&fitw=true' +
    '&sheetnames=false' +
    '&printtitle=false' +
    '&pagenumbers=false' +
    '&gridlines=false' +
    '&fzr=false';

  const token = ScriptApp.getOAuthToken();
  const blob = UrlFetchApp.fetch(pdfUrl, {
    headers: { Authorization: 'Bearer ' + token },
  }).getBlob().setName(fileBase + '.pdf');

  const file = folder.createFile(blob);
  const writeRow = appendLogLink_(file.getUrl(), fileBase);

  ui.alert('完了', `PDFを保存し、${APP_CONFIG.logSheetName}!E${writeRow} に記録しました。`, ui.ButtonSet.OK);
}

function saveDraftPdfAndQueue() {
  const ui = SpreadsheetApp.getUi();
  const sheet = getMainSheet_();
  const fields = getCurrentFieldValues_(sheet);
  const logs = getCurrentLogValues_(sheet);
  const requiredItems = [
    { key: 'workDate', label: '日付' },
    { key: 'berthNo', label: 'バースNo.' },
    { key: 'shipName', label: '船名' },
    { key: 'oilType', label: '油種' },
    { key: 'quantity', label: '数量' },
    { key: 'tankNo', label: '受入タンクNo.' },
    { key: 'category', label: '一般/黒物/機動' },
  ];
  const missing = requiredItems
    .filter((item) => !String(fields[item.key] || '').trim())
    .map((item) => item.label);

  if (missing.length) {
    ui.alert('入力不足のため中止', '次の項目が未入力です。\n\n・' + missing.join('\n・'), ui.ButtonSet.OK);
    return null;
  }

  const confirm = ui.alert('確認', '仮PDFを保存して承認待ち一覧に追加します。よろしいですか？', ui.ButtonSet.OK_CANCEL);
  if (confirm !== ui.Button.OK) return null;

  const reusedRecordId = resolveDraftQueueRecordId_(getCurrentQueueRecordId_(), {
    department: getSelectedDepartment_(),
    fields,
    logs,
  });
  ensureOilTypeOption_(fields.oilType);
  const file = exportSheetPdfToFolder_(getFolderByIdOrThrow_(APP_CONFIG.draftPdfFolderId), `${buildPdfBaseName_(fields)} 仮`);
  const record = upsertQueueRecord_({
    recordId: reusedRecordId,
    department: getSelectedDepartment_(),
    fields,
    logs,
    checks: getCheckboxStateMap_(sheet),
    stamps: getCurrentStampStatus_(sheet),
    draftPdfUrl: file.getUrl(),
    draftPdfName: file.getName(),
    status: 'staff_pending',
  });

  setCurrentQueueRecordId_(record.recordId);
  const writeRow = upsertLogLink_(record.recordId, file.getUrl(), file.getName(), '未完');
  if (!reusedRecordId) {
    appendTransferRow_({
      recordId: record.recordId,
      status: 'draft',
      department: getSelectedDepartment_(),
      fields,
      logs,
      pdfUrl: file.getUrl(),
      pdfName: file.getName(),
    });
  }
  ui.alert('完了', `仮PDFを保存し、案件ID ${record.recordId} を承認待ち一覧へ登録しました。\n${APP_CONFIG.logSheetName}!E${writeRow} にリンクも追加しました。`, ui.ButtonSet.OK);
  return getMainSidebarState();
}

function finalizeCurrentRecordPdf() {
  const ui = SpreadsheetApp.getUi();
  const recordId = getCurrentQueueRecordId_();
  if (!recordId) {
    ui.alert('案件未選択', '承認対象の案件を先に読み込んでください。', ui.ButtonSet.OK);
    return null;
  }

  const record = getQueueRecordById_(recordId);
  if (!record) throw new Error('案件が見つかりません。');
  if (record.status !== 'ready_final' && record.status !== 'done') {
    ui.alert('未完了', '主務職・課長の押印が完了してから確定PDFを保存してください。', ui.ButtonSet.OK);
    return null;
  }

  const sheet = getMainSheet_();
  const fields = getCurrentFieldValues_(sheet);
  const logs = getCurrentLogValues_(sheet);
  ensureOilTypeOption_(fields.oilType);
  const file = exportSheetPdfToFolder_(getFolderByIdOrThrow_(APP_CONFIG.finalPdfFolderId), `${buildPdfBaseName_(fields)} 確定版`);

  upsertQueueRecord_({
    recordId,
    department: getSelectedDepartment_(),
    fields,
    logs,
    checks: getCheckboxStateMap_(sheet),
    stamps: getCurrentStampStatus_(sheet),
    draftPdfUrl: record.draftPdfUrl,
    draftPdfName: record.draftPdfName,
    finalPdfUrl: file.getUrl(),
    finalPdfName: file.getName(),
    status: 'done',
  });
  upsertLogLink_(recordId, file.getUrl(), file.getName(), '完了');

  ui.alert('完了', `確定PDFを保存しました。\n案件ID: ${recordId}`, ui.ButtonSet.OK);
  return getMainSidebarState();
}

function finalizeCurrentRecordSilently_(recordId, sheet) {
  const id = String(recordId || '').trim();
  if (!id) return null;

  const record = getQueueRecordById_(id);
  if (!record) return null;

  const targetSheet = sheet || getMainSheet_();
  const fields = getCurrentFieldValues_(targetSheet);
  const logs = getCurrentLogValues_(targetSheet);
  ensureOilTypeOption_(fields.oilType);
  const file = exportSheetPdfToFolder_(getFolderByIdOrThrow_(APP_CONFIG.finalPdfFolderId), `${buildPdfBaseName_(fields)} 確定版`);

  upsertQueueRecord_({
    recordId: id,
    department: getSelectedDepartment_(),
    fields,
    logs,
    checks: getCheckboxStateMap_(targetSheet),
    stamps: getCurrentStampStatus_(targetSheet),
    draftPdfUrl: record.draftPdfUrl,
    draftPdfName: record.draftPdfName,
    finalPdfUrl: file.getUrl(),
    finalPdfName: file.getName(),
    status: 'done',
  });
  upsertLogLink_(id, file.getUrl(), file.getName(), '完了');
  return file;
}

function loadQueueRecord(recordId) {
  const id = String(recordId || '').trim();
  if (!id) throw new Error('読込対象の案件を選択してください。');

  const record = getQueueRecordById_(id);
  if (!record) throw new Error('案件が見つかりません。');

  setCurrentQueueRecordId_(record.recordId);
  clearOperationSheetSilently_();
  saveOperationRecord(record.payload);
  restoreQueueStamps_(record.payload.stamps || {});
  setCurrentQueueRecordId_(record.recordId);
  return getMainSidebarState();
}

function getMainSheet_() {
  return getSheetByNameOrThrow_(APP_CONFIG.mainSheetName);
}

function getSheetByNameOrThrow_(name) {
  const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(name);
  if (!sheet) throw new Error(`シート「${name}」が見つかりません。`);
  return sheet;
}

function getDepartments_() {
  const sheet = getMainSheet_();
  return sheet.getRange(APP_CONFIG.departments.labelsRange).getDisplayValues()[0]
    .map((label) => String(label || '').trim())
    .filter(String);
}

function getSelectedDepartment_() {
  const sheet = getMainSheet_();
  const labels = sheet.getRange(APP_CONFIG.departments.labelsRange).getDisplayValues()[0];
  const checks = sheet.getRange(APP_CONFIG.departments.checksRange).getValues()[0];
  for (let i = 0; i < checks.length; i += 1) {
    if (checks[i] === true) return String(labels[i] || '').trim();
  }
  return '';
}

function clearDepartmentSelection_(sheet) {
  sheet.getRange(APP_CONFIG.departments.checksRange).setValues([[false, false, false]]);
}

function applyDepartmentSelection_(sheet, department) {
  const value = String(department || '').trim();
  const labels = sheet.getRange(APP_CONFIG.departments.labelsRange).getDisplayValues()[0];
  const checks = labels.map((label) => String(label || '').trim() === value);
  sheet.getRange(APP_CONFIG.departments.checksRange).setValues([checks]);
}

function getAllRosterNames_() {
  const sheet = getSheetByNameOrThrow_(APP_CONFIG.rosterSheetName);
  const values = sheet.getDataRange().getDisplayValues();
  const seen = {};
  const names = [];

  for (let c = 0; c < values[0].length; c += 2) {
    for (let r = 1; r < values.length; r += 1) {
      const name = String(values[r][c + 1] || '').trim();
      if (!name || seen[name]) continue;
      seen[name] = true;
      names.push(name);
    }
  }
  return names;
}

function getRosterMap_() {
  const departments = getDepartments_();
  const map = {};
  departments.forEach((department) => {
    map[department] = getNamesByDepartment_(department);
  });
  return map;
}

function getNamesByDepartment_(department) {
  const selectedDepartment = String(department || '').trim();
  const sheet = getSheetByNameOrThrow_(APP_CONFIG.rosterSheetName);
  const values = sheet.getDataRange().getDisplayValues();
  const primary = [];
  const others = [];
  const seen = {};

  for (let c = 0; c < values[0].length; c += 2) {
    for (let r = 1; r < values.length; r += 1) {
      const dept = String(values[r][c] || '').trim();
      const name = String(values[r][c + 1] || '').trim();
      if (!name || seen[name]) continue;
      seen[name] = true;
      if (dept === selectedDepartment) primary.push(name);
      else others.push(name);
    }
  }
  return primary.concat(others);
}

function buildRosterState_(departments, selectedDepartment) {
  const sheet = getSheetByNameOrThrow_(APP_CONFIG.rosterSheetName);
  const values = sheet.getDataRange().getDisplayValues();
  const records = [];
  const seen = {};
  const departmentList = Array.isArray(departments) ? departments.slice() : [];

  if (!values.length) {
    return {
      allNames: [],
      namesByDepartment: [],
      byDepartment: {},
    };
  }

  for (let c = 0; c < values[0].length; c += 2) {
    for (let r = 1; r < values.length; r += 1) {
      const department = String(values[r][c] || '').trim();
      const name = String(values[r][c + 1] || '').trim();
      if (!name || seen[name]) continue;
      seen[name] = true;
      records.push({ department, name });
      if (department && departmentList.indexOf(department) === -1) {
        departmentList.push(department);
      }
    }
  }

  const allNames = records.map((record) => record.name);
  const byDepartment = {};
  departmentList.forEach((department) => {
    const primary = [];
    const others = [];
    records.forEach((record) => {
      if (record.department === department) primary.push(record.name);
      else others.push(record.name);
    });
    byDepartment[department] = primary.concat(others);
  });

  return {
    allNames,
    namesByDepartment: selectedDepartment ? (byDepartment[selectedDepartment] || []) : allNames,
    byDepartment,
  };
}

}

function readSnapshotCell_(snapshot, a1, type) {
  const ref = parseSingleCellA1_(a1);
  const rowIndex = ref.row - snapshot.startRow;
  const columnIndex = ref.column - snapshot.startColumn;
  if (rowIndex < 0 || columnIndex < 0) return '';

  const valueRow = snapshot.values[rowIndex];
  const displayRow = snapshot.displayValues[rowIndex];
  if (!valueRow || !displayRow) return '';

  const value = valueRow[columnIndex];
  const display = displayRow[columnIndex];
  if (value === '' || value === null) return '';

  if (type === 'date' && value instanceof Date) {
    return Utilities.formatDate(value, Session.getScriptTimeZone(), 'yyyy-MM-dd');
  }
  if (type === 'time') {
    return normalizeTimeText_(display);
  }
  return String(display || '').trim();
}

function parseSingleCellA1_(a1) {
  const match = String(a1 || '').trim().match(/^([A-Z]+)(\d+)$/i);
  if (!match) throw new Error(`Unsupported A1 notation: ${a1}`);
  return {
    column: a1ColumnToNumber_(match[1].toUpperCase()),
    row: Number(match[2]),
  };
}

function a1ColumnToNumber_(letters) {
  let column = 0;
  for (let i = 0; i < letters.length; i += 1) {
    column = (column * 26) + (letters.charCodeAt(i) - 64);
  }
  return column;
}

function readCellForUi_(range, type) {
  const value = range.getValue();
  if (value === '' || value === null) return '';

  if (type === 'date' && value instanceof Date) {
    return Utilities.formatDate(value, Session.getScriptTimeZone(), 'yyyy-MM-dd');
  }
  if (type === 'time') {
    return normalizeTimeText_(range.getDisplayValue());
  }
  return String(range.getDisplayValue() || '').trim();
}

function normalizeTimeText_(value) {
  const text = String(value == null ? '' : value).trim();
  if (!text) return '';
  const match = text.match(/^(\d{1,2}):(\d{2})/);
  if (!match) return text;
  return `${String(Number(match[1])).padStart(2, '0')}:${match[2]}`;
}

function hasInputValue_(value) {
  return String(value == null ? '' : value).trim() !== '';
}

function writeCellByType_(range, rawValue, type) {
  const value = String(rawValue == null ? '' : rawValue).trim();
  if (!value) {
    range.clearContent();
    return;
  }

  if (type === 'date') {
    const parts = parseSheetDateParts_(value);
    if (!parts) throw new Error(`日付形式が不正です: ${value}`);
    range.setFormula(`=DATE(${parts.year},${parts.month},${parts.day})`);
    return;
  }

  if (type === 'time') {
    const parts = parseSheetTimeParts_(value);
    if (!parts) throw new Error(`時刻形式が不正です: ${value}`);
    range.setFormula(`=TIME(${parts.hour},${parts.minute},0)`);
    range.setNumberFormat('hh:mm');
    return;
  }

  if (type === 'number') {
    const numeric = parseLooseNumber_(value);
    if (!Number.isFinite(numeric)) throw new Error(`数値形式が不正です: ${value}`);
    range.setValue(numeric);
    return;
  }

  range.setValue(value);
}

function parseSheetDateParts_(value) {
  const match = String(value || '').trim().match(/^(\d{4})-(\d{2})-(\d{2})$/);
  if (!match) return null;
  const year = Number(match[1]);
  const month = Number(match[2]);
  const day = Number(match[3]);
  const date = new Date(year, month - 1, day);
  if (
    date.getFullYear() !== year ||
    date.getMonth() !== month - 1 ||
    date.getDate() !== day
  ) {
    return null;
  }
  return { year, month, day };
}

function parseSheetTimeParts_(value) {
  const text = String(value || '').trim();
  if (!text) return null;

  let hour;
  let minute;
  let match = text.match(/^(\d{1,2}):(\d{2})$/);
  if (match) {
    hour = Number(match[1]);
    minute = Number(match[2]);
  } else {
    match = text.replace(/[^0-9]/g, '').match(/^(\d{3,4})$/);
    if (!match) return null;
    const digits = match[1].padStart(4, '0');
    hour = Number(digits.slice(0, 2));
    minute = Number(digits.slice(2, 4));
  }

  if (!Number.isInteger(hour) || !Number.isInteger(minute)) return null;
  if (hour < 0 || hour > 23 || minute < 0 || minute > 59) return null;
  return { hour, minute };
}

function parseLooseNumber_(value) {
  const text = String(value == null ? '' : value).trim();
  if (!text) return NaN;
  const normalized = text
    .replace(/,/g, '')
    .replace(/[Ａ-Ｚａ-ｚ０-９．－]/g, (char) => String.fromCharCode(char.charCodeAt(0) - 65248))
    .replace(/[^\d.\-+]/g, '');
  if (!normalized || normalized === '+' || normalized === '-' || normalized === '.' || normalized === '+.' || normalized === '-.') {
    return NaN;
  }
  return Number(normalized);
}

function getCheckboxStateMap_(sheet) {
  const range = sheet.getRange(APP_CONFIG.pdfRangeA1);
  const validations = range.getDataValidations();
  const states = {};

  for (let row = 0; row < validations.length; row += 1) {
    for (let column = 0; column < validations[row].length; column += 1) {
      const rule = validations[row][column];
      if (!rule || rule.getCriteriaType() !== SpreadsheetApp.DataValidationCriteria.CHECKBOX) continue;
      const cell = sheet.getRange(range.getRow() + row, range.getColumn() + column);
      const a1 = cell.getA1Notation();
      states[a1] = cell.isChecked() === true;
    }
  }
  return states;
}

function restoreCheckboxStateMap_(sheet, states) {

function moveImageToCell_(sheet, image, targetA1) {
  const target = sheet.getRange(targetA1);
  image.setAnchorCell(target);
  image.setAnchorCellXOffset(0);
  image.setAnchorCellYOffset(0);

}

function fitImageToCell_(sheet, image, cellRange) {
  const targetRange = getImageFitRange_(cellRange);
  const col = targetRange.getColumn();
  const row = targetRange.getRow();
  const maxW = Math.max(1, getRangePixelWidth_(sheet, targetRange) - 4);
  const maxH = Math.max(1, getRangePixelHeight_(sheet, targetRange) - 4);
  const width = image.getWidth();
  const height = image.getHeight();
  const scale = Math.min(maxW / Math.max(width, 1), maxH / Math.max(height, 1));
  const newW = Math.max(1, Math.floor(width * scale));
  const newH = Math.max(1, Math.floor(height * scale));

  image.setWidth(newW).setHeight(newH);
  image.setAnchorCellXOffset(Math.max(0, Math.floor((getRangePixelWidth_(sheet, targetRange) - newW) / 2)));
  image.setAnchorCellYOffset(Math.max(0, Math.floor((getRangePixelHeight_(sheet, targetRange) - newH) / 2)));
}

function getImageFitRange_(cellRange) {
  const a1 = cellRange.getA1Notation();
  const slotKeys = Object.keys(APP_CONFIG.stampSlots);
  for (let i = 0; i < slotKeys.length; i += 1) {
    const slot = APP_CONFIG.stampSlots[slotKeys[i]];
    if (slot.anchorCell === a1 && slot.fitRangeA1) {
      return cellRange.getSheet().getRange(slot.fitRangeA1);
    }
  }
  const merged = cellRange.getMergedRanges();
  return merged && merged.length ? merged[0] : cellRange;
}

function getRangePixelWidth_(sheet, range) {
  let total = 0;
  for (let column = range.getColumn(); column < range.getColumn() + range.getNumColumns(); column += 1) {
    total += sheet.getColumnWidth(column);
  }
  return total;
}

function getRangePixelHeight_(sheet, range) {
  let total = 0;
  for (let row = range.getRow(); row < range.getRow() + range.getNumRows(); row += 1) {
    const rowHeight = sheet.getRowHeight(row);
    const isHidden = sheet.isRowHiddenByUser(row) || rowHeight <= 0;
    total += isHidden ? APP_CONFIG.hiddenStampFitRowHeight : rowHeight;
  }
  return total;
}

function ensureColumns_(sheet, requiredCols) {

  if (current < requiredCols) sheet.insertColumnsAfter(current, requiredCols - current);
}

function appendLogLink_(url, linkText) {
  const logSs = SpreadsheetApp.openById(APP_CONFIG.logSpreadsheetId);
  const logSheet = logSs.getSheetByName(APP_CONFIG.logSheetName);
  if (!logSheet) throw new Error(`ログシート「${APP_CONFIG.logSheetName}」が見つかりません。`);

  const startRow = APP_CONFIG.logStartRow;
  const col = APP_CONFIG.logColumn;
  const lastRow = Math.max(logSheet.getLastRow(), startRow);
  const values = logSheet.getRange(startRow, col, lastRow - startRow + 1, 1).getDisplayValues();

  let writeRow = startRow;
  for (let i = values.length - 1; i >= 0; i -= 1) {
    if (String(values[i][0] || '').trim()) {
      writeRow = startRow + i + 1;
      break;
    }
  }

  const safeText = String(linkText).replace(/"/g, '""');
  logSheet.getRange(writeRow, col).setFormula(`=HYPERLINK("${url}","${safeText}")`);
  return writeRow;
}

function upsertLogLink_(recordId, url, linkText, statusLabel) {
  const logSs = SpreadsheetApp.openById(APP_CONFIG.logSpreadsheetId);
  const logSheet = logSs.getSheetByName(APP_CONFIG.logSheetName);
  if (!logSheet) throw new Error(`ログシート「${APP_CONFIG.logSheetName}」が見つかりません。`);

  const writeRow = findLogRowByRecordId_(logSheet, recordId) || findNextLogRow_(logSheet);
  const linkCell = logSheet.getRange(writeRow, APP_CONFIG.logColumn);
  const statusCell = logSheet.getRange(writeRow, APP_CONFIG.logColumn + 1);
  const safeText = String(linkText || '').replace(/"/g, '""');
  const status = String(statusLabel || '').trim();

  linkCell.setFormula(`=HYPERLINK("${url}","${safeText}")`);
  if (status === '完了') linkCell.clearNote();
  else linkCell.setNote(String(recordId || '').trim());
  statusCell.setValue(status);
  return writeRow;
}

function findLogRowByRecordId_(sheet, recordId) {
  const id = String(recordId || '').trim();
  if (!id) return 0;

  const startRow = APP_CONFIG.logStartRow;
  const lastRow = Math.max(sheet.getLastRow(), startRow);
  for (let row = startRow; row <= lastRow; row += 1) {
    if (String(sheet.getRange(row, APP_CONFIG.logColumn).getNote() || '').trim() === id) {
      return row;
    }
  }
  return 0;
}

function clearLogLinkByRecordId_(recordId) {
  const id = String(recordId || '').trim();
  if (!id) return;

  const logSs = SpreadsheetApp.openById(APP_CONFIG.logSpreadsheetId);
  const logSheet = logSs.getSheetByName(APP_CONFIG.logSheetName);
  if (!logSheet) return;

  const rowIndex = findLogRowByRecordId_(logSheet, id);
  if (!rowIndex) return;

  logSheet.getRange(rowIndex, APP_CONFIG.logColumn, 1, 2).clearContent();
  logSheet.getRange(rowIndex, APP_CONFIG.logColumn).clearNote();
}

function findNextLogRow_(sheet) {
  const startRow = APP_CONFIG.logStartRow;
  const lastRow = Math.max(sheet.getLastRow(), startRow);
  const values = sheet.getRange(startRow, APP_CONFIG.logColumn, lastRow - startRow + 1, 1).getDisplayValues();

  let writeRow = startRow;
  for (let i = values.length - 1; i >= 0; i -= 1) {
    if (String(values[i][0] || '').trim()) {
      writeRow = startRow + i + 1;
      break;
    }
  }
  return writeRow;
}

function appendTransferRow_(data) {
  const sheet = getTransferSheet_();
  const row = buildTransferRow_(data || {});
  const writeRow = findNextTransferRow_(sheet);
  sheet.getRange(writeRow, 1, 1, row.length).setValues([row]);
  return writeRow;
}

function getTransferSheet_() {
  const ss = SpreadsheetApp.openById(APP_CONFIG.transferSpreadsheetId);
  const sheet = ss.getSheetByName(APP_CONFIG.transferSheetName);

function findNextTransferRow_(sheet) {
  const startRow = APP_CONFIG.transferStartRow;
  const lastRow = Math.max(sheet.getLastRow(), startRow);
  const values = sheet.getRange(startRow, 2, lastRow - startRow + 1, 1).getDisplayValues();

  for (let i = 0; i < values.length; i += 1) {
    if (!String(values[i][0] || '').trim()) {
      return startRow + i;
    }
  }
  return lastRow + 1;
}

function buildTransferRow_(data) {
  const fields = data.fields || {};
  const logs = Array.isArray(data.logs) ? data.logs : [];
  const requiredHours = buildRequiredMinutes_(fields.startVolume, fields.endVolume);
  return [
    String(fields.workDate || '').trim(),
    1,
    String(fields.berthNo || '').trim(),
    String(fields.shipName || '').trim(),
    String(fields.oilType || '').trim(),
    String(fields.quantity || '').trim(),
    String(fields.tankNo || '').trim(),
    buildPumpUsageLabel_(fields),
    String(fields.dischargePressure || '').trim(),
    String(fields.pierPressure || '').trim(),
    buildPeakFlowRate_(fields, logs),
    buildAverageFlowRate_(fields.quantity, requiredHours),
    requiredHours,
  ];
}

function buildPumpUsageLabel_(fields) {
  const capacity = String(fields.pumpCapacity || '').trim();
  const count = String(fields.pumpCount || '').trim();
  if (capacity && count) return `${capacity} × ${count}台`;
  return capacity || count || '';
}

function buildPeakFlowRate_(fields, logs) {
  const candidates = []
    .concat((logs || []).map((entry) => entry && entry.flow))
    .concat([fields.bridgeFlowRate]);
  const numbers = candidates
    .map((value) => parseLooseNumber_(value))
    .filter((value) => Number.isFinite(value));
  if (!numbers.length) return '';
  return String(Math.max.apply(null, numbers));
}

function buildRequiredMinutes_(startValue, endValue) {
  const start = parseSheetTimeParts_(startValue);
  const end = parseSheetTimeParts_(endValue);
  if (!start || !end) return '';

  const startMinutes = (start.hour * 60) + start.minute;
  const endMinutes = (end.hour * 60) + end.minute;
  let diff = endMinutes - startMinutes;
  if (diff < 0) diff += 24 * 60;
  const hours = Math.floor(diff / 60);
  const minutes = diff % 60;
  return `${hours}:${String(minutes).padStart(2, '0')}`;
}

function buildLogSummary_(logs) {
  return logs
    .filter((entry) => {
      if (!entry) return false;
      return ['time', 'pressure', 'flow', 'amount', 'person']
        .some((key) => String(entry[key] || '').trim());
    })
    .map((entry) => {
      const parts = [];
      if (entry.time) parts.push(`time:${entry.time}`);
      if (entry.person) parts.push(`person:${entry.person}`);
      if (entry.pressure) parts.push(`pressure:${entry.pressure}`);
      if (entry.flow) parts.push(`flow:${entry.flow}`);
      if (entry.amount) parts.push(`amount:${entry.amount}`);
      return parts.join(', ');
    })
    .join(' | ');
}

function resolveDraftQueueRecordId_(currentRecordId, payload) {
  const recordId = String(currentRecordId || '').trim();
  if (!recordId) return '';

  const record = getQueueRecordById_(recordId);
  if (!record) return '';
  if (record.status === 'done') return '';

  return areQueuePayloadsEquivalent_(record.payload, payload) ? recordId : '';
}

function areQueuePayloadsEquivalent_(left, right) {
  const normalize = (value) => JSON.stringify({
    department: String(value && value.department || '').trim(),
    fields: value && value.fields || {},
    logs: Array.isArray(value && value.logs) ? value.logs : [],
  });
  return normalize(left) === normalize(right);
}

function getQueueSummaries_() {
  const sheet = ensureQueueSheet_();
  const lastRow = sheet.getLastRow();
  if (lastRow < 2) return [];

  const rows = sheet.getRange(2, 1, lastRow - 1, sheet.getLastColumn()).getDisplayValues();
  const summaries = rows
    .filter((row) => String(row[0] || '').trim())
    .filter((row) => isPendingQueueStatus_(row[1]))
    .map((row) => ({
      recordId: row[0],
      status: getApprovalStatusLabel_(row[1]),
      nextApprover: getNextApproverLabel_(row[1]),
      title: row[2],
      workDate: row[3],
      shipName: row[4],
      berthNo: row[5],
      department: row[6],
      draftPdfUrl: row[7],
      finalPdfUrl: row[9],
      updatedAt: row[13],
    }));

  summaries.sort((a, b) => String(b.updatedAt).localeCompare(String(a.updatedAt)));
  return summaries.slice(0, 30);
}

function getRequiredStampSlot_(status) {

function upsertQueueRecord_(data) {
  const sheet = ensureQueueSheet_();
  const recordId = String(data.recordId || generateQueueRecordId_()).trim();
  const payload = {
    department: String(data.department || '').trim(),
    fields: data.fields || {},
    logs: Array.isArray(data.logs) ? data.logs : [],
    checks: data.checks || {},
    stamps: data.stamps || {},
  };
  const now = Utilities.formatDate(new Date(), Session.getScriptTimeZone(), 'yyyy-MM-dd HH:mm:ss');
  const title = buildPdfBaseName_(payload.fields);
  const rowValues = [
    recordId,
    String(data.status || 'staff_pending').trim(),
    title,
    payload.fields.workDate || '',
    payload.fields.shipName || '',
    payload.fields.berthNo || '',
    payload.department,
    data.draftPdfUrl || '',
    data.draftPdfName || '',
    data.finalPdfUrl || '',
    data.finalPdfName || '',
    JSON.stringify(payload),
    '',
    now,
    data.status === 'done' ? now : '',
  ];

  const existingRow = findQueueRowById_(sheet, recordId);
  if (existingRow) {
    const createdAt = sheet.getRange(existingRow, 13).getDisplayValue() || now;
    rowValues[12] = createdAt;
    if (data.status !== 'done') rowValues[14] = sheet.getRange(existingRow, 15).getDisplayValue();
    sheet.getRange(existingRow, 1, 1, rowValues.length).setValues([rowValues]);
  } else {
    rowValues[12] = now;
    sheet.appendRow(rowValues);
  }

  return {
    recordId,
    payload,
    draftPdfUrl: rowValues[7],
    draftPdfName: rowValues[8],
    finalPdfUrl: rowValues[9],
    finalPdfName: rowValues[10],
    status: rowValues[1],
    title,
  };
}

function getQueueRecordById_(recordId) {
  const sheet = ensureQueueSheet_();
  const rowIndex = findQueueRowById_(sheet, recordId);
  if (!rowIndex) return null;

  const row = sheet.getRange(rowIndex, 1, 1, sheet.getLastColumn()).getValues()[0];
  return {
    recordId: row[0],
    status: row[1],
    title: row[2],
    workDate: row[3],
    shipName: row[4],
    berthNo: row[5],
    department: row[6],
    draftPdfUrl: row[7],
    draftPdfName: row[8],
    finalPdfUrl: row[9],
    finalPdfName: row[10],
    payload: row[11] ? JSON.parse(row[11]) : { department: '', fields: {}, logs: [], stamps: {} },
    createdAt: row[12],
    updatedAt: row[13],
    finalizedAt: row[14],
  };
}

      'createdAt', 'updatedAt', 'finalizedAt',

    sheet.setFrozenRows(1);
    sheet.hideSheet();
  }
  return sheet;
}

    }
  }
  return 0;
}

function saveDraftPdfAndQueue() {

  const logs = getCurrentLogValues_(sheet);
  const requiredItems = [
    { key: 'workDate', label: '日付' },
    { key: 'berthNo', label: 'バースNo.' },

    { key: 'oilType', label: '油種' },

  ];

    .filter((item) => !String(fields[item.key] || '').trim())
    .map((item) => item.label);

    ui.alert('入力不足のため中止', '次の項目が未入力です。\n\n・' + missing.join('\n・'), ui.ButtonSet.OK);

    recordId: reusedRecordId,
    department: getSelectedDepartment_(),

    logs,
    checks: getCheckboxStateMap_(sheet),
    stamps: getCurrentStampStatus_(sheet),
    draftPdfUrl: file.getUrl(),
    draftPdfName: file.getName(),
    status: 'staff_pending',
  });

  setCurrentQueueRecordId_(record.recordId);
  const writeRow = upsertLogLink_(record.recordId, file.getUrl(), file.getName(), '未完');
  if (!reusedRecordId) {
    appendTransferRow_({
      recordId: record.recordId,
      status: 'draft',
      department: getSelectedDepartment_(),
      fields,
      logs,
      pdfUrl: file.getUrl(),
      pdfName: file.getName(),

function trashDriveFileByUrl_(url) {
  const id = extractDriveFileId_(url);
  if (!id) return;
  try {
    DriveApp.getFileById(id).setTrashed(true);
  } catch (error) {
    // Ignore missing or already-trashed files.
  }

function extractDriveFileId_(url) {
  const value = String(url || '').trim();
  if (!value) return '';
  const byPath = value.match(/\/d\/([a-zA-Z0-9_-]+)/);
  if (byPath) return byPath[1];
  const byQuery = value.match(/[?&]id=([a-zA-Z0-9_-]+)/);
  return byQuery ? byQuery[1] : '';
}

function buildPumpUsageLabel_(fields) {
  const capacity = String(fields.pumpCapacity || '').trim();
  const count = String(fields.pumpCount || '').trim();
  if (capacity && count) return `${capacity} × ${count}台`;
  return capacity || count || '';
}

function buildRequiredMinutes_(startValue, endValue) {
  const start = parseSheetTimeParts_(startValue);
  const end = parseSheetTimeParts_(endValue);
  if (!start || !end) return '';

  const startMinutes = (start.hour * 60) + start.minute;
  const endMinutes = (end.hour * 60) + end.minute;
  let diff = endMinutes - startMinutes;
  if (diff < 0) diff += 24 * 60;
  return (Math.round((diff / 60) * 10) / 10).toFixed(1);
}

function buildAverageFlowRate_(quantityValue, requiredHoursValue) {
  const quantity = parseLooseNumber_(quantityValue);
  const requiredHours = parseLooseNumber_(requiredHoursValue);
  if (!Number.isFinite(quantity) || !Number.isFinite(requiredHours) || requiredHours <= 0) return '';
  return (Math.round((quantity / requiredHours) * 10) / 10).toFixed(1);
}

function upsertQueueRecord_(data) {
  const sheet = ensureQueueSheet_();
  const recordId = String(data.recordId || generateQueueRecordId_()).trim();
  const payload = {
    department: String(data.department || '').trim(),
    fields: data.fields || {},
    logs: Array.isArray(data.logs) ? data.logs : [],
    checks: data.checks || {},
    stamps: data.stamps || {},
  };
  const now = Utilities.formatDate(new Date(), Session.getScriptTimeZone(), 'yyyy-MM-dd HH:mm:ss');
  const title = buildPdfBaseName_(payload.fields);
  const rowValues = [
    recordId,
    String(data.status || 'staff_pending').trim(),
    title,
    payload.fields.workDate || '',
    payload.fields.shipName || '',
    payload.fields.berthNo || '',
    payload.department,
    data.draftPdfUrl || '',
    data.draftPdfName || '',
    data.finalPdfUrl || '',
    data.finalPdfName || '',
    JSON.stringify(payload),
    '',
    now,
    data.status === 'done' ? now : '',
  ];

  const existingRow = findQueueRowById_(sheet, recordId);
  if (existingRow) {
    const createdAt = sheet.getRange(existingRow, 13).getDisplayValue() || now;
    rowValues[12] = createdAt;
    if (data.status !== 'done') rowValues[14] = sheet.getRange(existingRow, 15).getDisplayValue();
    sheet.getRange(existingRow, 1, 1, rowValues.length).setValues([rowValues]);
  } else {
    rowValues[12] = now;
    sheet.appendRow(rowValues);
  }

  const record = {
    recordId,
    payload,
    draftPdfUrl: rowValues[7],
    draftPdfName: rowValues[8],
    finalPdfUrl: rowValues[9],
    finalPdfName: rowValues[10],
    status: rowValues[1],
    title,
    updatedAt: rowValues[13],
  };
  syncQueueRecordToExternalTargets_(record);
  return record;
}

  syncCaseIdRecord_(record);

function syncStatusBoardRecord_(record) {
  const ss = SpreadsheetApp.openById(APP_CONFIG.logSpreadsheetId);
  const sheet = ss.getSheetByName('主務職');
  if (!sheet) return;

  const rowIndex = findStatusBoardRow_(sheet, record.recordId);
  const writeRow = rowIndex || findNextStatusBoardRow_(sheet);
  const values = [[
    `=HYPERLINK("${buildApprovalSpreadsheetUrl_(record.recordId)}","${record.recordId}")`,
    buildStatusBoardCaseLabel_(record),
    getRoleBoardStatus_('chief', record.status),
    getRoleBoardStatus_('staff', record.status),

  ]];
  sheet.getRange(writeRow, 1, 1, values[0].length).setValues(values);
}

  const ss = SpreadsheetApp.openById(APP_CONFIG.logSpreadsheetId);
  const sheet = ss.getSheetByName('主務職');

  const rowIndex = findStatusBoardRow_(sheet, recordId);

}

function findStatusBoardRow_(sheet, recordId) {
  const startRow = 6;
  const lastRow = sheet.getLastRow();
  if (lastRow < startRow) return 0;
  const formulas = sheet.getRange(startRow, 1, lastRow - startRow + 1, 1).getFormulas();
  const displays = sheet.getRange(startRow, 1, lastRow - startRow + 1, 1).getDisplayValues();
  for (let i = 0; i < displays.length; i += 1) {
    const visible = String(displays[i][0] || '').trim();
    const formula = String(formulas[i][0] || '');
    if (visible === recordId || formula.indexOf(recordId) !== -1) return startRow + i;
  }
  return 0;
}

function findNextStatusBoardRow_(sheet) {
  const startRow = 6;
  const lastRow = Math.max(sheet.getLastRow(), startRow);
  const values = sheet.getRange(startRow, 1, lastRow - startRow + 1, 1).getDisplayValues();
  for (let i = 0; i < values.length; i += 1) {
    if (!String(values[i][0] || '').trim()) return startRow + i;
  }
  return lastRow + 1;
}

function getRoleBoardStatus_(role, status) {

  return '未';
}

function syncCaseIdRecord_(record) {
  if (!record || String(record.status || '').trim() !== 'done') return;

  const recordId = String(record.recordId || '').trim();
  if (!recordId) return;

  const ss = SpreadsheetApp.openById(APP_CONFIG.caseIdSpreadsheetId);
  const sheet = getCaseIdSheet_(ss);
  if (!sheet) throw new Error(`案件ID転記先シート「${APP_CONFIG.caseIdSheetName}」が見つかりません。`);

  const finalPdfUrl = String(record.finalPdfUrl || '').trim();
  const finalPdfName = String(record.finalPdfName || '').trim() || '確定済みPDF';
  const pdfDisplay = finalPdfUrl ? `=HYPERLINK("${finalPdfUrl.replace(/"/g, '""')}","${finalPdfName.replace(/"/g, '""')}")` : '';
  const dataStartRow = getCaseIdDataStartRow_(sheet);

  const rowIndex = findCaseIdRowByRecordId_(sheet, recordId);
  if (rowIndex > 0) {
    sheet.getRange(rowIndex, 1, 1, 2).setValues([[recordId, pdfDisplay]]);
    return;
  }

  const writeRow = findNextCaseIdRow_(sheet, dataStartRow);
  sheet.getRange(writeRow, 1, 1, 2).setValues([[recordId, pdfDisplay]]);
}

function findCaseIdRowByRecordId_(sheet, recordId) {
  const startRow = getCaseIdDataStartRow_(sheet);
  const lastRow = sheet.getLastRow();
  if (lastRow < startRow) return 0;

  const values = sheet.getRange(startRow, 1, lastRow - startRow + 1, 1).getDisplayValues();
  for (let i = 0; i < values.length; i += 1) {
    if (String(values[i][0] || '').trim() === String(recordId || '').trim()) {
      return startRow + i;
    }
  }
  return 0;
}

function getCaseIdSheet_(ss) {
  const targetId = Number(APP_CONFIG.caseIdSheetId || 0);
  if (targetId > 0) {
    const byId = ss.getSheets().find((sheet) => sheet.getSheetId() === targetId);
    if (byId) return byId;
  }
  return ss.getSheetByName(APP_CONFIG.caseIdSheetName);
}

function getCaseIdDataStartRow_(sheet) {
  const maxScan = Math.min(sheet.getLastRow(), 200);
  if (maxScan < 1) return 2;
  const values = sheet.getRange(1, 1, maxScan, 2).getDisplayValues();
  let headerRow = 0;
  for (let i = 0; i < values.length; i += 1) {
    const colA = String(values[i][0] || '').trim();
    const colB = String(values[i][1] || '').trim();
    if (colA === '案件ID' && colB === '案件内容') headerRow = i + 1;
  }
  if (headerRow > 0) return headerRow + 1;

  for (let i = values.length - 1; i >= 0; i -= 1) {
    if (String(values[i][0] || '').trim() === '案件ID') return i + 2;
  }
  return 2;
}

function findNextCaseIdRow_(sheet, startRow) {
  const begin = Math.max(Number(startRow || 2), 2);
  const lastRow = Math.max(sheet.getLastRow(), begin);
  const values = sheet.getRange(begin, 1, lastRow - begin + 1, 1).getDisplayValues();
  for (let i = 0; i < values.length; i += 1) {
    if (!String(values[i][0] || '').trim()) return begin + i;
  }
  return lastRow + 1;
}

function ensureQueueSheetInSpreadsheet_(ss) {
  let sheet = ss.getSheetByName(APP_CONFIG.queueSheetName);
  if (!sheet) {
    sheet = ss.insertSheet(APP_CONFIG.queueSheetName);
    sheet.getRange(1, 1, 1, 15).setValues([[
      'recordId', 'status', 'title', 'workDate', 'shipName', 'berthNo', 'department',
      'draftPdfUrl', 'draftPdfName', 'finalPdfUrl', 'finalPdfName', 'payloadJson',
      'createdAt', 'updatedAt', 'finalizedAt',
    ]]);
    sheet.setFrozenRows(1);
    sheet.hideSheet();
  }
  return sheet;
}

function setupRecommendedProtections() {
  const applied = [];
  const mainSheet = getMainSheet_();
  const mainUnprotectedRanges = buildMainSheetEditableRanges_(mainSheet);
  upsertSheetProtection_(mainSheet, 'Codex Main Input Protection', mainUnprotectedRanges);
  applied.push(`${mainSheet.getName()}: 入力セルとチェック欄のみ編集可`);

  [
    APP_CONFIG.rosterSheetName,
    APP_CONFIG.stampMapSheetName,
    APP_CONFIG.dropdownSettingsSheetName,
    APP_CONFIG.queueSheetName,
  ].forEach((name) => {
    const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(name);
    if (!sheet) return;
    upsertSheetProtection_(sheet, `Codex Protected ${name}`, []);
    applied.push(`${name}: シート全体を保護`);
  });

  SpreadsheetApp.getUi().alert(
    '保護設定を適用しました',
    applied.join('\n'),
    SpreadsheetApp.getUi().ButtonSet.OK
  );
}

function buildMainSheetEditableRanges_(sheet) {
  const ranges = [];
  const seen = {};

  APP_CONFIG.clearRanges.forEach((a1) => {
    const key = String(a1 || '').trim();
    if (!key || seen[key]) return;
    seen[key] = true;
    ranges.push(sheet.getRange(key));
  });

  collectCheckboxCells_(sheet).forEach((cell) => {
    const key = cell.getA1Notation();
    if (seen[key]) return;
    seen[key] = true;
    ranges.push(cell);
  });

  return ranges;
}

function collectCheckboxCells_(sheet) {
  const range = sheet.getRange(APP_CONFIG.pdfRangeA1);
  const validations = range.getDataValidations();
  const cells = [];

  for (let row = 0; row < validations.length; row += 1) {
    for (let col = 0; col < validations[row].length; col += 1) {
      const rule = validations[row][col];
      if (!rule) continue;
      if (rule.getCriteriaType() !== SpreadsheetApp.DataValidationCriteria.CHECKBOX) continue;
      cells.push(range.getCell(row + 1, col + 1));
    }
  }

  return cells;
}

function upsertSheetProtection_(sheet, description, unprotectedRanges) {
  const protections = sheet.getProtections(SpreadsheetApp.ProtectionType.SHEET);
  let protection = protections.find((item) => item.getDescription() === description);
  if (!protection) {
    protection = sheet.protect();
    protection.setDescription(description);
  }

  protection.setWarningOnly(false);
  protection.setUnprotectedRanges(unprotectedRanges || []);

  try {
    if (protection.canDomainEdit()) protection.setDomainEdit(false);
  } catch (error) {
    // Domain editing may be unavailable depending on the account type.
  }

  return protection;
}
